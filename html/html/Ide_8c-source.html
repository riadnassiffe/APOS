00060         int count, status;
00061         short info[256];
00062         int IdeStatus = 0;
00063 
00064         /*
00065          * Caso não tenha sido detectado nenhum HD ele tenta detectar o principal(Master), caso algum já tenha sido detectado,
00066          *ele tenta detecar o escravo(slave).
00067          */
00068         if (diskNumber == 0) {
00069                 //enviamos para aos registradores do dispositivo, o código correspondente a HD master.
00070                 Out_Byte(IDE_REG_DEVICE_HEAD, IDE_Master);
00071         } else {
00072                 //enviamos para aos registradores do dispositivo, o código correspondente a HD slave.
00073                 Out_Byte(IDE_REG_DEVICE_HEAD, IDE_Slave);
00074         }
00075         //enviamos para o dispositivo, responsável de executar a ações, um código para ele tentar reconhecer algum dispositivo,
00076         // que neste momento já deve ter seu código no registrador esperado pelo commando.
00077         Out_Byte(IDE_STATUS_REGISTER,IDE_COMMAND_IDENTIFY_DRIVE);
00078         
00079         IdeStatus = diskWait();
00080         
00081         /*
00082          *Antes de dar o HD como reconhecido, precisamos verificar se a operação ocorreu com sucesso ou não, para isso temos que verificar o valor armazenado no 
00083          * registrador IDE_STATUS_REGISTER(0x1f7)
00084          */ 
00085         status = In_Byte(IDE_STATUS_REGISTER);
00086 
00087         //Se o valor armazenado for 0, significa que o dispositivo foi reconhecido, com sucesso, caso o contrario não existe ou pode esta com algum problema.
00088         if (IdeStatus == 0) {
00089                 //caso hd tenha sido reconhecido com sucesso, precisamos agora ir no registrador onde ele guarda as informações e ler elas.
00090                 for (count=0; count < 256; count++) {
00091                         info[count]= In_Word(IDE_REG_DEVICE_DATA_PORT);
00092                 }
00093 
00094                 //a operação traz um vetor com 256 posições, sendo que no momento somente 4 delas nos importam, que são as seguintes:
00095                 diskList[diskNumber].cylinders=info[IDE_INDENTIFY_CYLINDERS];
00096                 diskList[diskNumber].heads=info[IDE_INDENTIFY_HEADERS];
00097                 diskList[diskNumber].sectorPerTrack=info[IDE_INDENTIFY_SECTORS_TRACK];
00098                 diskList[diskNumber].bytesPerSector=info[IDE_INDENTIFY_BYTES_SECTOR];
00099                 diskList[diskNumber].status = 0;
00100                 getDiskInfo(diskNumber);
00101 
00102         }else{
00103                 return EXIT_FAILURE;
00104         }
00105 
00106         return EXIT_SUCCESS;
00107 }
